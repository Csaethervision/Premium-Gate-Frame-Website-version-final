        function updateEstimate() {
            const category = document.getElementById('productCategory').value;
            const size = document.getElementById('productSize').value;
            const quantity = parseInt(document.getElementById('productQuantity').value) || 0;
            const estimateDiv = document.getElementById('priceEstimate');
            
            if (category && size && quantity && productPrices[category] && productPrices[category][size]) {
                const unitPrice = productPrices[category][size];
                const total = unitPrice * quantity;
                estimateDiv.innerHTML = `Estimated Total: <span class="text-green-400">KSh ${total.toLocaleString()}</span>`;
                estimateDiv.classList.add('price-glow');
            } else {
                estimateDiv.textContent = 'Select products to see estimate';
                estimateDiv.classList.remove('price-glow');
            }
        }

        const cursorRing = cursor.querySelector('.cursor-ring');
        const cursorShine = cursor.querySelector('.cursor-shine');

        let mouseX = 0;
        let mouseY = 0;
        let cursorX = 0;
        let cursorY = 0;
        let shineX = 0;

        document.addEventListener('mousemove', function(e) {

        function animateCursor() {
            const dotSpeed = 1;
            const ringSpeed = 0.15;
            const shineSpeed = 0.08;
            
            cursorX += (mouseX - cursorX) * ringSpeed;
            cursorY += (mouseY - cursorY) * ringSpeed;
            shineX += (mouseX - shineX) * shineSpeed;
            shineY += (mouseY - shineY) * shineSpeed;
            
            cursorDot.style.left = mouseX + 'px';
            cursorDot.style.top = mouseY + 'px';
            cursorRing.style.left = cursorX + 'px';
            cursorRing.style.top = cursorY + 'px';
            cursorShine.style.left = shineX + 'px';
            cursorShine.style.top = shineY + 'px';
            
            requestAnimationFrame(animateCursor);
        }
        animateCursor();
